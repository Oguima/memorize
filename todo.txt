.font(Font.largeTitle)


struct CardView: View {
    var isFaceUp: Bool

    var body: some view {
        ZStack {
            if isFaceUp {
                RoundedRectangle(cornerRadius: 10.0).fill(Color.white)
                RoundedRectangle(cornerRadius: 10.0).stroke(lineWidth: 3)
                Text("X")
            } else {
                RoundedRectangle(cornerRadius: 10.0).fill()
            }
        }
    }
}
